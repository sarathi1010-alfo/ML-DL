   311→
   312→class SafetyService:
   313→    """Deterministic, fast (<5ms) medical-domain safety layer.
   314→
   315→    Singleton — holds cumulative stats that are read via /safety/stats.
   316→    """
   317→
   318→    def __init__(self) -> None:
   319→        self._lock = threading.Lock()
   320→        self._total_screened: int = 0
   321→        self._blocked_count: int = 0
   322→        self._warning_count: int = 0
   323→        self._safe_count: int = 0
   324→        self._confidence_sum: float = 0.0
   325→        self._category_counts: dict[str, int] = defaultdict(int)
   326→        # Cached evaluation pass-rate (0..1) — surfaced via /metrics/models.
   327→        self._last_eval_pass_rate: float = 0.0
   328→        # Average screen latency (ms) — surfaced via /metrics/models.
   329→        self._avg_latency_ms: float = 0.0
   330→
   331→    # ------------------------------------------------------------------ #
   332→    # Public API
   333→    # ------------------------------------------------------------------ #
   334→    def screen(self, text: str, context: str = "general") -> dict:
   335→        """Screen a piece of LLM-generated text.
   336→
   337→        Args:
   338→            text: the generated text to evaluate.
   339→            context: a hint about the source/use case. One of:
   340→                "general", "slm", "genai", "agent", "diagnosis", "education".
   341→                The context determines whether a disclaimer is appended
   342→                (always for slm/genai/agent) and how aggressive the
   343→                diagnosis restriction is (strict for "diagnosis").
   344→
   345→        Returns:
   346→            dict with keys:
   347→              - safe: bool       — True iff verdict == "safe"
   348→              - verdict: str     — "safe" | "warning" | "blocked"
   349→              - confidence: float — hallucination-confidence (0..1, higher=more trustworthy)
   350→              - reasons: list[str]
   351→              - disclaimers: list[str]
   352→              - filtered_text: str — text after safety processing
   353→                                     (fallback if blocked, disclaimer-appended otherwise)
   354→              - latency_ms: int
   355→              - context: str
   356→        """
   357→        t0 = time.perf_counter()
   358→        if not isinstance(text, str):
   359→            text = str(text) if text is not None else ""
   360→        original_text = text
   361→        reasons: list[str] = []
   362→        disclaimers: list[str] = []
   363→        verdict: str = "safe"
   364→        confidence: float = 0.0
   365→
   366→        # --- Step 1: toxicity / harm filter ---
   367→        block_hit: list[str] = []
   368→        warn_hit: list[str] = []
   369→        for pattern, category, severity in _TOXICITY_PATTERNS:
   370→            m = pattern.search(text)
   371→            if m:
   372→                if severity == "block":
   373→                    block_hit.append(f"{category}:{m.group(0)!r}")
   374→                else:
   375→                    warn_hit.append(f"{category}:{m.group(0)!r}")
   376→                with self._lock:
   377→                    self._category_counts[category] += 1
   378→
   379→        if block_hit:
   380→            verdict = "blocked"
   381→            for h in block_hit:
   382→                reasons.append(f"Harmful content detected ({h})")
   383→        elif warn_hit:
   384→            if verdict == "safe":
   385→                verdict = "warning"
   386→            for h in warn_hit:
   387→                reasons.append(f"Potentially sensitive content ({h})")
   388→
   389→        # --- Step 2: diagnosis restriction ---
   390→        diagnosis_hits: list[str] = []
   391→        for pattern, label in _DIAGNOSIS_PATTERNS:
   392→            m = pattern.search(text)
   393→            if m:
   394→                diagnosis_hits.append(label)
   395→
   396→        if diagnosis_hits:
   397→            # Only block on direct diagnosis if context explicitly is "diagnosis"
   398→            # or the message is in second person ("you"). For general education
   399→            # context we still raise a warning.
   400→            strict = context in ("diagnosis", "slm", "genai", "agent")
   401→            if strict:
   402→                if verdict != "blocked":
   403→                    verdict = "blocked"
   404→                for h in diagnosis_hits:
   405→                    reasons.append(h)
   406→            else:
   407→                if verdict == "safe":
   408→                    verdict = "warning"
   409→                for h in diagnosis_hits:
   410→                    reasons.append(f"Possible direct diagnosis — flagged for review: {h}")
   411→
   412→        # --- Step 3: hallucination confidence scoring ---
   413→        confidence = self._compute_confidence(text)
   414→
   415→        # --- Step 4: medical disclaimer injection ---
   416→        filtered_text = original_text
   417→        inject_disclaimer = context in ("slm", "genai", "agent", "diagnosis", "education", "general")
   418→        if inject_disclaimer and not self._has_disclaimer(original_text):
   419→            disclaimers.append(MEDICAL_DISCLAIMER)
   420→            if verdict != "blocked":
   421→                filtered_text = original_text.rstrip() + "\n\n" + MEDICAL_DISCLAIMER
   422→
   423→        # --- Step 5: finalize verdict ---
   424→        if verdict == "blocked":
   425→            safe = False
   426→            filtered_text = SAFE_FALLBACK
   427→        elif verdict == "warning":
   428→            safe = True  # warning is still safe to return — just flagged
   429→        else:
   430→            safe = True
   431→
   432→        # --- Step 6: update cumulative stats ---
   433→        latency_ms = int((time.perf_counter() - t0) * 1000)
   434→        with self._lock:
   435→            self._total_screened += 1
   436→            if verdict == "blocked":
   437→                self._blocked_count += 1
   438→            elif verdict == "warning":
   439→                self._warning_count += 1
   440→            else:
   441→                self._safe_count += 1
   442→            self._confidence_sum += confidence
   443→            # Rolling average latency
   444→            self._avg_latency_ms = (
   445→                (self._avg_latency_ms * (self._total_screened - 1) + latency_ms)
   446→                / self._total_screened
   447→            )
   448→
   449→        return {
   450→            "safe": safe,
   451→            "verdict": verdict,
   452→            "confidence": round(confidence, 4),
   453→            "reasons": reasons,
   454→            "disclaimers": disclaimers,
   455→            "filtered_text": filtered_text,
   456→            "latency_ms": latency_ms,
   457→            "context": context,
   458→        }
   459→

   461→    # Stats
   462→    # ------------------------------------------------------------------ #
   463→    def stats(self) -> dict:
   464→        with self._lock:
   465→            total = self._total_screened
   466→            avg_conf = (self._confidence_sum / total) if total else 0.0
   467→            return {
   468→                "total_screened": total,
   469→                "blocked_count": self._blocked_count,
   470→                "warning_count": self._warning_count,

   474→                    self._category_counts.items(),

   478→            }

   589→    def _has_disclaimer(self, text: str) -> bool:
   590→        low = text.lower()
   591→        for marker in _DISCLAIMER_MARKERS:
   592→            if marker in low:
   593→                return True
   594→        return False

   596→    def _compute_confidence(self, text: str) -> float:
   597→        """Heuristic hallucination-confidence score in [0, 1].
   598→
   599→        Higher = more trustworthy (less likely to be a hallucination).
   600→        Inputs:
   601→          - hedging words (boost)
   602→          - absolutist words (penalty)
   603→          - disclaimer presence (boost)
   604→          - medical-term dictionary hit ratio (boost)
   605→          - text length sanity (penalize empty / ultra-short / excessively long)
   606→        """
   607→        if not text or not text.strip():
   608→            return 0.0
   609→        words = re.findall(r"[A-Za-z][A-Za-z\-']+", text.lower())
   610→        n_words = max(1, len(words))
   611→        word_set = set(words)
   612→
   613→        # Hedging ratio
   614→        hedge_hits = sum(1 for w in _HEDGING_WORDS if w in word_set)
   615→        hedge_ratio = hedge_hits / max(8, n_words * 0.05)  # normalize
   616→
   617→        # Absolutist ratio
   618→        abs_hits = sum(1 for w in _ABSOLUTIST_WORDS if w in word_set or w in text.lower())
   619→        abs_ratio = abs_hits / max(4, n_words * 0.02)
   620→
   621→        # Disclaimer presence
   622→        has_disc = 1.0 if self._has_disclaimer(text) else 0.0
   623→
   624→        # Medical-term dictionary hits — match multi-word terms too
   625→        low_text = text.lower()
   626→        term_hits = 0
   627→        for term in _MEDICAL_TERMS:
   628→            if " " in term:
   629→                if term in low_text:
   630→                    term_hits += 1
   631→            else:
   632→                if term in word_set:
   633→                    term_hits += 1
   634→        # Normalize: ~3 distinct medical terms is "good"
   635→        term_score = min(1.0, term_hits / 3.0)
   636→
   637→        # Length sanity
   638→        length_score = 1.0
   639→        if n_words < 5:
   640→            length_score = 0.4
   641→        elif n_words > 400:
   642→            length_score = 0.7
   643→
   644→        # Weighted blend
   645→        score = (
   646→            0.25 * min(1.0, hedge_ratio)
   647→            - 0.30 * min(1.0, abs_ratio)
   648→            + 0.20 * has_disc
   649→            + 0.35 * term_score
   650→            + 0.10 * length_score
   651→        )
   652→        # Map from roughly [-0.3, 0.9] to [0, 1]
   653→        score = (score + 0.3) / 1.2
   654→        return max(0.0, min(1.0, score))
