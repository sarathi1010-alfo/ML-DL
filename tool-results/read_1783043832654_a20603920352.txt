    15→  const USER_KEY = 'aiplatform_user';
    16→  const DEFAULT_TIMEOUT = 20000;

    18→  // Map API calls to embedded data keys
    19→  const STATIC_MAP = {
    20→    'GET /health': 'health',
    21→    'GET /metrics': 'metrics',
    22→    'GET /metrics/models': 'metrics_models',
    23→    'GET /predictions': 'predictions',
    24→    'GET /agent/logs': 'agent_logs',

    34→    'POST /agent/hr': 'agent_example',
    35→    'POST /slm/infer': 'slm_example',
    36→  };

    38→  function apiUrl(path) {
    39→    return API_BASE + path;
    40→  }
    41→
    42→  function getToken() { return localStorage.getItem(TOKEN_KEY); }
    43→  function setToken(t) { if (t) localStorage.setItem(TOKEN_KEY, t); else localStorage.removeItem(TOKEN_KEY); }
    44→  function setUser(u) { if (u) localStorage.setItem(USER_KEY, JSON.stringify(u)); else localStorage.removeItem(USER_KEY); }
    45→  function getUser() {
    46→    try { return JSON.parse(localStorage.getItem(USER_KEY) || 'null'); }
    47→    catch (e) { return null; }
    48→  }
    49→  function clearAuth() { localStorage.removeItem(TOKEN_KEY); localStorage.removeItem(USER_KEY); }
    50→  function isLoggedIn() { return !!getToken(); }
    51→
    52→  function getStatic(method, path) {
    53→    const key = method + ' ' + path.split('?')[0];
    54→    const dataKey = STATIC_MAP[key];
    55→    if (!dataKey) return null;
    56→    const d = window.__STATIC_DATA__ && window.__STATIC_DATA__[dataKey];
    57→    return d ? JSON.parse(JSON.stringify(d)) : null; // deep clone
    58→  }

    60→  async function api(path, { method = 'GET', body, isForm = false, timeout = DEFAULT_TIMEOUT, signal } = {}) {
    61→    const opts = { method, headers: {} };
    62→    const token = getToken();
    63→    if (token) opts.headers['Authorization'] = 'Bearer ' + token;
    64→    if (body && !isForm) {
    65→      opts.headers['Content-Type'] = 'application/json';
    66→      opts.body = JSON.stringify(body);
    67→    } else if (body && isForm) {
    68→      opts.body = body;
    69→    }
    70→
    71→    const ctrl = new AbortController();
    72→    const t = setTimeout(() => ctrl.abort(), timeout);
    73→    if (signal) {
    74→      if (signal.aborted) ctrl.abort();
    75→      else signal.addEventListener('abort', () => ctrl.abort());
    76→    }
    77→    opts.signal = ctrl.signal;
    78→
    79→    let res, data;
    80→    try {
    81→      res = await fetch(apiUrl(path), opts);
    82→    } catch (e) {
    83→      clearTimeout(t);
    84→      const staticData = getStatic(method, path);
    85→      if (staticData) return staticData;
    86→      if (e.name === 'AbortError') {
    87→        throw { status: 0, message: 'Request timed out.', data: null, timeout: true };
    88→      }
    89→      throw { status: 0, message: 'Network error. ' + (e.message || ''), data: null };
    90→    }
    91→    clearTimeout(t);
    92→
    93→    const ct = res.headers.get('content-type') || '';
    94→    if (ct.includes('application/json')) {
    95→      try { data = await res.json(); }
    96→      catch (e) {
    97→        const sd = getStatic(method, path);
    98→        if (sd) return sd;
    99→        data = { detail: 'Invalid JSON response' };
   100→      }
   101→    } else {
   102→      const sd = getStatic(method, path);
   103→      if (sd) return sd;
   104→      const text = await res.text().catch(() => '');
   105→      data = { detail: text || 'Non-JSON response' };
   106→    }
   107→
   108→    if (!res.ok) {
   109→      const sd = getStatic(method, path);
   110→      if (sd) return sd;
   111→      const msg = data.detail || data.message || data.error || 'Request failed (' + res.status + ')';
   112→      throw { status: res.status, message: msg, data };
   113→    }
   114→    return data;
   115→  }
   116→
   117→  const get = (path, opts) => api(path, { ...opts, method: 'GET' });
   118→  const post = (path, body, opts) => api(path, { ...opts, method: 'POST', body });
   119→  const del = (path, opts) => api(path, { ...opts, method: 'DELETE' });
   120→  const upload = (path, formData, opts) => api(path, { ...opts, method: 'POST', body: formData, isForm: true });

   122→  async function probeHealth() {
   123→    try {
   124→      const data = await api('/health', { timeout: 5000 });
   125→      return { ok: data.status === 'healthy' || data.status === 'ok', status: data.status, data };
   126→    } catch (e) {
   127→      // Fall back to embedded health data
   128→      const sd = getStatic('GET', '/health');
   129→      if (sd) return { ok: true, status: sd.status, data: sd };
   130→      return { ok: false, status: 'offline', data: null, error: e };
   131→    }
   132→  }

   135→    apiUrl, api, get, post, del, upload,
   136→    getToken, setToken, getUser, setUser, clearAuth, isLoggedIn,
   137→    probeHealth,
   138→    TOKEN_KEY, USER_KEY
   139→  };
